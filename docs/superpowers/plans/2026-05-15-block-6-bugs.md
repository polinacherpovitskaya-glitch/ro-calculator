## Task 1: SQL

```sql
-- 006_bug_reports.sql
CREATE TABLE IF NOT EXISTS bug_reports (
    id              BIGINT PRIMARY KEY,
    title           TEXT NOT NULL,
    description     TEXT,
    severity        TEXT CHECK (severity IN ('low','medium','high','critical')),
    status          TEXT NOT NULL DEFAULT 'open' CHECK (status IN ('open','in_progress','fixed','wontfix','duplicate')),
    page            TEXT,
    reporter_email  TEXT,
    reporter_name   TEXT,
    assignee_id     INTEGER REFERENCES employees(id) ON DELETE SET NULL,
    extras          JSONB DEFAULT '{}'::jsonb,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    fixed_at        TIMESTAMPTZ
);

CREATE TABLE IF NOT EXISTS bug_attachments (
    id              BIGSERIAL PRIMARY KEY,
    bug_id          BIGINT NOT NULL REFERENCES bug_reports(id) ON DELETE CASCADE,
    filename        TEXT NOT NULL,
    mime_type       TEXT,
    size_bytes      BIGINT,
    storage_key     TEXT NOT NULL,          -- ключ в Selectel S3 (без https://)
    uploaded_at     TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS idx_bug_attachments_bug ON bug_attachments(bug_id);

INSERT INTO app_meta (id, version) VALUES (1, '006-bugs')
ON CONFLICT (id) DO UPDATE SET version = EXCLUDED.version, applied_at = NOW();
```

## Task 2: S3 helper

**File:** `ops/api/src/s3.js`

```js
// Тонкая обёртка над @aws-sdk/client-s3 для Selectel Object Storage.
import { S3Client, PutObjectCommand, GetObjectCommand, DeleteObjectCommand } from '@aws-sdk/client-s3';
import { getSignedUrl } from '@aws-sdk/s3-request-presigner';

const client = new S3Client({
  endpoint: process.env.S3_ENDPOINT,
  region: 'ru-1',
  credentials: {
    accessKeyId: process.env.S3_ACCESS_KEY,
    secretAccessKey: process.env.S3_SECRET_KEY,
  },
  forcePathStyle: true,
});
const BUCKET = process.env.S3_BUCKET;

export async function uploadObject(key, body, contentType) {
  await client.send(new PutObjectCommand({ Bucket: BUCKET, Key: key, Body: body, ContentType: contentType }));
}

export async function deleteObject(key) {
  await client.send(new DeleteObjectCommand({ Bucket: BUCKET, Key: key }));
}

export async function presignedGetUrl(key, expiresIn = 600) {
  return getSignedUrl(client, new GetObjectCommand({ Bucket: BUCKET, Key: key }), { expiresIn });
}
```

Установить пакеты в `ops/api/package.json`: `@aws-sdk/client-s3`, `@aws-sdk/s3-request-presigner`, `multer` (для multipart upload).

## Task 4: Миграция файлов из Supabase Storage в Selectel

**File:** `ops/scripts/migrate-storage-bug-attachments.mjs`

```js
// Перенос файлов из bucket bug-attachments (Supabase) в Selectel.
// Для каждого файла: download → upload → обновить storage_key в bug_attachments.
import { createClient } from '@supabase/supabase-js';
import { uploadObject } from '../api/src/s3.js';
import pg from 'pg';

const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_SERVICE_KEY);
const pool = new pg.Pool({ connectionString: process.env.DATABASE_URL });

async function main() {
  // Получить список всех вложений в БД
  const { rows } = await pool.query(
    `SELECT id, storage_key FROM bug_attachments WHERE storage_key LIKE 'supabase://%'`
  );
  console.log(`Migrating ${rows.length} attachments`);

  for (const att of rows) {
    const sbKey = att.storage_key.replace('supabase://', '');
    const { data, error } = await supabase.storage.from('bug-attachments').download(sbKey);
    if (error) { console.error(`Skip ${att.id}:`, error.message); continue; }
    const buffer = Buffer.from(await data.arrayBuffer());
    const newKey = `bug-attachments/${att.id}-${sbKey.split('/').pop()}`;
    await uploadObject(newKey, buffer, data.type);
    await pool.query(`UPDATE bug_attachments SET storage_key = $1 WHERE id = $2`, [newKey, att.id]);
    console.log(`✓ ${att.id} → ${newKey}`);
  }

  await pool.end();
}

main().catch(err => { console.error(err); process.exit(1); });
```

Важно: эта миграция запускается **один раз перед cutover**. Во время Stage A файлы остаются в Supabase (читаем оттуда через старые URL), а копию в Selectel делаем за 24 часа до cutover.

Альтернативно: сразу при копировании bug_reports скачиваем и заливаем в Selectel — выбрать что проще.

## Task 5: Selectel Object Storage bucket setup

**Manual step.** В Selectel панели:
1. Object Storage → Создать контейнер
2. Имя: `ro-ops-bugs-attachments` (или общий `ro-ops-storage` с папками)
3. Тип: Приватный
4. Регион: ru-1

Создать service user `ops-app` с правами на этот bucket. Получить access_key + secret_key.

Сохранить в `/srv/ops/infra/.env` на VPS:
```
S3_ENDPOINT=https://s3.storage.selcloud.ru
S3_BUCKET=ro-ops-bugs-attachments
S3_ACCESS_KEY=...
S3_SECRET_KEY=...
```
